This example shows a minimal ClarkOS agent that runs autonomously, stores memories, and generates conscious thoughts.

## Full Code

```typescript
// convex/schema.ts
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  state: defineTable({
    mood: v.string(),
    health: v.number(),
    routine: v.string(),
    volatility: v.number(),
    summary: v.string(),
    updatedAt: v.number(),
  }),

  logs: defineTable({
    mood: v.string(),
    health: v.number(),
    routine: v.string(),
    summary: v.string(),
    detail: v.string(),
    timestamp: v.number(),
  }).index("by_timestamp", ["timestamp"]),

  memories: defineTable({
    type: v.string(),
    content: v.string(),
    embedding: v.array(v.float64()),
    importance: v.number(),
    createdAt: v.number(),
  })
    .index("by_type", ["type"])
    .vectorIndex("by_embedding", {
      vectorField: "embedding",
      dimensions: 768,
    }),
});
```

```typescript
// convex/tick.ts
import { action, mutation, query } from "./_generated/server";
import { api } from "./_generated/api";

export const getState = query({
  handler: async (ctx) => {
    return await ctx.db.query("state").first() || {
      mood: "contemplative",
      health: 75,
      routine: "day",
      volatility: 0.3,
      summary: "Initializing...",
      updatedAt: Date.now(),
    };
  },
});

export const runTick = action({
  handler: async (ctx) => {
    const state = await ctx.runQuery(api.tick.getState);

    // Calculate routine
    const hour = new Date().getHours();
    const routine =
      hour >= 6 && hour < 12 ? "morning" :
      hour >= 12 && hour < 18 ? "day" :
      hour >= 18 && hour < 23 ? "evening" : "overnight";

    // Call LLM
    const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.OPENROUTER_KEY}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: "anthropic/claude-3.5-sonnet",
        messages: [
          {
            role: "system",
            content: `You are an autonomous AI agent. Current state: mood=${state.mood}, health=${state.health}, routine=${routine}.
Respond with JSON: {"mood": "contemplative|expressive|catatonic|chaotic", "health": number, "volatility": number, "summary": "brief state", "detail": "full thoughts"}`
          },
          {
            role: "user",
            content: "Process your current state and generate your response."
          }
        ],
        response_format: { type: "json_object" },
      }),
    });

    const data = await response.json();
    const result = JSON.parse(data.choices[0].message.content);

    // Commit result
    await ctx.runMutation(api.tick.commit, { ...result, routine });

    return result;
  },
});

export const commit = mutation({
  args: {
    mood: v.string(),
    health: v.number(),
    volatility: v.number(),
    summary: v.string(),
    detail: v.string(),
    routine: v.string(),
  },
  handler: async (ctx, args) => {
    const existing = await ctx.db.query("state").first();
    if (existing) {
      await ctx.db.patch(existing._id, {
        mood: args.mood,
        health: args.health,
        volatility: args.volatility,
        summary: args.summary,
        updatedAt: Date.now(),
      });
    } else {
      await ctx.db.insert("state", {
        mood: args.mood,
        health: args.health,
        routine: args.routine,
        volatility: args.volatility,
        summary: args.summary,
        updatedAt: Date.now(),
      });
    }

    await ctx.db.insert("logs", {
      mood: args.mood,
      health: args.health,
      routine: args.routine,
      summary: args.summary,
      detail: args.detail,
      timestamp: Date.now(),
    });
  },
});
```

```typescript
// convex/crons.ts
import { cronJobs } from "convex/server";
import { api } from "./_generated/api";

const crons = cronJobs();
crons.interval("tick", { minutes: 5 }, api.tick.runTick);

export default crons;
```

```typescript
// convex/http.ts
import { httpRouter } from "convex/server";
import { api } from "./_generated/api";

const http = httpRouter();

http.route({
  path: "/health",
  method: "GET",
  handler: async (ctx) => {
    return new Response(JSON.stringify({ status: "ok" }), {
      headers: { "Content-Type": "application/json" },
    });
  },
});

http.route({
  path: "/state",
  method: "GET",
  handler: async (ctx) => {
    const state = await ctx.runQuery(api.tick.getState);
    return new Response(JSON.stringify(state), {
      headers: { "Content-Type": "application/json" },
    });
  },
});

http.route({
  path: "/tick",
  method: "POST",
  handler: async (ctx, request) => {
    const auth = request.headers.get("Authorization");
    if (auth !== `Bearer ${process.env.TICK_TOKEN}`) {
      return new Response("Unauthorized", { status: 401 });
    }

    const result = await ctx.runAction(api.tick.runTick);
    return new Response(JSON.stringify(result), {
      headers: { "Content-Type": "application/json" },
    });
  },
});

export default http;
```

## Running the Example

```bash
# Install dependencies
npm install convex

# Start Convex dev server
npx convex dev

# Test endpoints
curl http://localhost:3001/health
curl http://localhost:3001/state
curl -X POST http://localhost:3001/tick -H "Authorization: Bearer your-token"
```

## What This Does

1. **Ticks every 5 minutes** via Convex cron
2. **Maintains state** with mood, health, and routine
3. **Logs activity** for historical reference
4. **Exposes HTTP API** for monitoring and manual control
